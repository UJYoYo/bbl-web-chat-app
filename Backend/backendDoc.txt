Install:
- Spring boot extension
- Java minimum v22

Re-build XML File:
    mvn clean install
Run Sprint Boot:
    mvn spring-boot:run

Controllers:
- HomeController
    - Endpoints:
        - GET:
            - /hello 
        - POST:
            - /register 
            - Description: register / login user
                - Parameter
                    userBody: username, password

- FriendController
    - Endpoints:
        - GET:
            - /getUser 
                - Description: Search if user exists and returns user
                - Parameter
                    - username
            - /getRequests 
                - Description: Get add friend requests for a user
                - Parameter
                        - username
        - POST:
            - /sendRequest
                - Description: Send a friend request after searching for a user
                - Parameter
                    sendRequestBody: recipientUsername, senderUsername, status
            - /requests
                - Description: Accept or reject a friend request
                - Parameter
                    sendRequestBody: recipientUsername, senderUsername, status

- MessageController
    - Endpoints:
        - GET:
            - /getHistory 
                - Parameter
                    - roomId
        - POST:
            - /sendMessage
                - Parameter
                    messageBody: senderId, recipientId, message
